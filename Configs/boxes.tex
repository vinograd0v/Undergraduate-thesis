\newcommand{\hugequote}{%
	{\fontsize{75}{75}\selectfont%
	\hspace*{-.560em}\color{ctcolorgraylight}%
	\textit{\glqq}%
	\vskip -.26em}%
}

\newcommand{\cleanchapterquote}[3]{%
	\begin{minipage}{.865\textwidth}%
		\begin{flushright}
			\begin{minipage}{.65\textwidth}%
				\begin{flushleft}
					{\hugequote}\textit{#1}
				\end{flushleft}
	    		\begin{flushright}
    				--- \textbf{#2} \\
    				#3
		    	\end{flushright}
		    \end{minipage}%
		\end{flushright}
	\end{minipage}%
	\bigskip
}
